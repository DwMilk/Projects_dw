- 创建父子进程，在子进程中向管道写入“hello world”，在父进程读出

```c++
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>
#include <sys/wait.h>

#define BUFFER_SIZE 20

int main() {
    int pipefd[2];  // 管道文件描述符数组
    pid_t pid;
    char buffer[BUFFER_SIZE];

    // 创建管道
    if (pipe(pipefd) == -1) {
        perror("pipe");
        exit(1);
    }

    // 创建子进程
    pid = fork();

    if (pid < 0) {  // 错误处理
        perror("fork");
        exit(1);
    } else if (pid == 0) {  // 子进程
        close(pipefd[0]);  // 关闭读端

        // 向管道写入数据
        const char* message = "hello world";
        write(pipefd[1], message, strlen(message) + 1);
        
        close(pipefd[1]);  // 写完后关闭写端
        exit(0);
    } else {  // 父进程
        close(pipefd[1]);  // 关闭写端

        // 等待子进程结束
        wait(NULL);

        // 从管道读取数据
        ssize_t bytes_read = read(pipefd[0], buffer, sizeof(buffer) - 1);
        if (bytes_read > 0) {
            buffer[bytes_read] = '\0';  // 确保字符串以null结尾
            printf("父进程读取到：%s\n", buffer);
        } else {
            perror("read");
        }

        close(pipefd[0]);  // 读完后关闭读端
    }

    return 0;
}
```



- 编写一个程序，捕捉信号SIGINT，从键盘输入ctrl+c之后显示“捕捉到中断信号，程序退出”

```c++
#include <stdio.h>
#include <stdlib.h>
#include <signal.h>
#include <unistd.h>

// 信号处理函数
void sigint_handler(int sig) {
    printf("\n捕捉到中断信号，程序退出\n");
    exit(0);
}

int main() {
    // 注册 SIGINT 信号的处理函数
    if (signal(SIGINT, sigint_handler) == SIG_ERR) {
        perror("无法捕获 SIGINT");
        exit(1);
    }

    printf("程序运行中，按 Ctrl+C 发送中断信号...\n");

    // 保持程序运行，直到接收到信号
    while(1) {
        sleep(1);
    }

    return 0;
}
```

- 编写一个包含2个线程的程序，在主线程创建一个全局变量并初始化为0，在另一个线程对应这个全局变量进行递加计算，并在结束时想主线程返回一个结果，由主线程打印输出。

```c
#include <stdio.h>
#include <stdlib.h>
#include <pthread.h>
#include <unistd.h>

// 全局变量
int global_var = 0;

// 线程函数
void* increment_thread(void* arg) {
    int local_count = 0;
    for (int i = 0; i < 1000000; i++) {
        global_var++;
        local_count++;
    }
    
    // 将局部计数作为结果返回给主线程
    int* result = malloc(sizeof(int));
    *result = local_count;
    return (void*)result;
}

int main() {
    pthread_t thread_id;
    void* thread_result;

    // 创建新线程
    if (pthread_create(&thread_id, NULL, increment_thread, NULL) != 0) {
        perror("pthread_create failed");
        exit(1);
    }

    // 等待线程结束并获取返回值
    if (pthread_join(thread_id, &thread_result) != 0) {
        perror("pthread_join failed");
        exit(1);
    }

    // 从线程获取结果并释放内存
    int increment_count = *(int*)thread_result;
    free(thread_result);

    // 打印结果
    printf("线程实际递增次数: %d\n", increment_count);
    printf("全局变量最终值: %d\n", global_var);

    return 0;
}
```



- 编写一个程序，打开一个文本文件，读取其中内容，将其复制到一个新建文件中

```c++
#include <iostream>
#include <fstream>
#include <string>

void copyFile(const std::string& sourceFile, const std::string& destinationFile) {
    std::ifstream source(sourceFile, std::ios::binary);
    if (!source) {
        std::cerr << "错误：无法打开源文件 " << sourceFile << std::endl;
        return;
    }

    std::ofstream destination(destinationFile, std::ios::binary);
    if (!destination) {
        std::cerr << "错误：无法创建目标文件 " << destinationFile << std::endl;
        return;
    }

    destination << source.rdbuf();

    if (source.fail() || destination.fail()) {
        std::cerr << "错误：复制过程中发生错误" << std::endl;
    } else {
        std::cout << "文件内容已从 " << sourceFile << " 成功复制到 " << destinationFile << std::endl;
    }

    source.close();
    destination.close();
}

int main() {
    std::string sourceFile = "source.txt";      // 请替换为实际的源文件名
    std::string destinationFile = "destination.txt";  // 请替换为实际的目标文件名

    copyFile(sourceFile, destinationFile);

    return 0;
}
```



- 用锁和条件变量两个线程交替打印A和B

```c++
#include <iostream>
#include <thread>
#include <mutex>
#include <condition_variable>

class Printer {
private:
    std::mutex mtx;
    std::condition_variable cv;
    bool isA = true;  // 用于控制打印A还是B

public:
    void printA() {
        for (int i = 0; i < 10; ++i) {  // 打印10次
            std::unique_lock<std::mutex> lock(mtx);
            cv.wait(lock, [this]{ return isA; });
            std::cout << "A" << std::flush;
            isA = false;
            cv.notify_one();
        }
    }

    void printB() {
        for (int i = 0; i < 10; ++i) {  // 打印10次
            std::unique_lock<std::mutex> lock(mtx);
            cv.wait(lock, [this]{ return !isA; });
            std::cout << "B" << std::flush;
            isA = true;
            cv.notify_one();
        }
    }
};

int main() {
    Printer printer;
    std::thread t1(&Printer::printA, &printer);
    std::thread t2(&Printer::printB, &printer);

    t1.join();
    t2.join();

    std::cout << std::endl;
    return 0;
}
```
